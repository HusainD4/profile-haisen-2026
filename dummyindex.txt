      // 1. SPA Router Logic
      function navigateTo(pageId, scrollToId = null) {
        // Sembunyikan semua halaman
        document.querySelectorAll('.page-section').forEach(p => {
          p.style.display = 'none';
          // Reset animasi agar bisa diputar ulang
          p.querySelectorAll('[data-animate]').forEach(el => {
            el.classList.remove('animate-fade-in-up', 'animate-fade-in-left', 'animate-fade-in-right');
            el.classList.add('opacity-0');
          });
        });

        // Tampilkan halaman tujuan
        const targetPage = document.getElementById('page-' + pageId);
        targetPage.style.display = 'block';
        window.scrollTo(0, 0);

        // Trigger ulang animasi scroll untuk halaman baru
        setTimeout(() => {
          const observer = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
              if (entry.isIntersecting) {
                const el = entry.target;
                const anim = el.getAttribute('data-animate');
                if(anim) {
                  el.classList.add('animate-' + anim);
                  observer.unobserve(el);
                }
              }
            });
          }, { threshold: 0.1 });
          targetPage.querySelectorAll('[data-animate]').forEach(el => observer.observe(el));
        }, 50);

        // Jika ada request scroll ke section tertentu
        if (scrollToId) {
          setTimeout(() => {
            document.getElementById(scrollToId).scrollIntoView({ behavior: 'smooth' });
          }, 100);
        }

        closeMenu(); // Tutup mobile menu jika terbuka
      }

      function goToSection(sectionId) {
        const homePage = document.getElementById('page-home');
        if (homePage.style.display === 'none') {
          navigateTo('home', sectionId);
        } else {
          document.getElementById(sectionId).scrollIntoView({ behavior: 'smooth' });
        }
        closeMenu();
      }
